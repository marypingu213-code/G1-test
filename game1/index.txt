<script>
    // 題目資料：不含解釋，僅有題目、選項、答案
    const questions = [
        { text: "我和爸爸去方子裡。", options: ["防", "房", "坊"], answer: "房" },
        { text: "為十麼你不好好上課。", options: ["什", "神", "加"], answer: "什" },
        { text: "青幫我把課本拿出來。", options: ["請", "情", "精"], answer: "請" },
        { text: "也是一個可愛的小朋友。", options: ["拖", "尾", "他"], answer: "他" },
        { text: "都是一隻小兔子。", options: ["部", "陳", "那"], answer: "那" },
        { text: "大加要一起去玩。", options: ["豕", "家", "宅"], answer: "家" },
        { text: "你真是太友福氣了。", options: ["有", "月", "朋"], answer: "有" },
        { text: "兔子不喜歡被泡著。", options: ["包", "抱", "泡"], answer: "抱" }
    ];

    let currentQ = [];
    let index = 0;
    let score = 0;

    function initGame() {
        // 隨機選 5 題
        currentQ = [...questions].sort(() => 0.5 - Math.random()).slice(0, 5);
        index = 0;
        score = 0;
        document.getElementById('game-screen').style.display = 'block';
        document.getElementById('result-screen').style.display = 'none';
        updateUI();
        loadQuestion();
    }

    function loadQuestion() {
        const q = currentQ[index];
        document.getElementById('question-text').innerText = q.text;
        const optsDiv = document.getElementById('options-area');
        optsDiv.innerHTML = '';
        document.getElementById('next-btn').style.display = 'none';

        q.options.forEach(opt => {
            const btn = document.createElement('button');
            btn.className = 'btn';
            btn.innerText = opt;
            btn.onclick = () => check(btn, opt, q.answer);
            optsDiv.appendChild(btn);
        });
    }

    function check(btn, selection, correct) {
        const btns = document.querySelectorAll('.options .btn');
        btns.forEach(b => b.disabled = true); // 鎖定按鈕

        if (selection === correct) {
            btn.classList.add('correct');
            score += 20;
        } else {
            btn.classList.add('wrong');
            // 顯示正確答案
            btns.forEach(b => {
                if (b.innerText === correct) b.classList.add('correct');
            });
        }
        
        updateUI();
        document.getElementById('next-btn').style.display = 'inline-block';
    }

    function nextQuestion() {
        index++;
        if (index < currentQ.length) {
            loadQuestion();
            updateUI();
        } else {
            showResult();
        }
    }

    function updateUI() {
        document.getElementById('score').innerText = `得分: ${score}`;
        document.getElementById('progress').innerText = `題數: ${Math.min(index + 1, currentQ.length)}/${currentQ.length}`;
    }

    function showResult() {
        document.getElementById('game-screen').style.display = 'none';
        document.getElementById('result-screen').style.display = 'block';
        document.getElementById('final-score').innerText = `${score} 分`;
    }

    initGame();
</script>
